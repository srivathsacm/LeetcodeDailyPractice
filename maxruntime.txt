class Solution {
    public long maxRunTime(int n, int[] batteries) {
        long sum = 0;
        for (int b : batteries) sum += b;
        long low = 0, high = sum / n;
        while (low < high) {
            long mid = (low + high + 1) >> 1;
            long power = 0;
            for (int b : batteries) power += Math.min(b, mid);
            if (power >= mid * n) low = mid;
            else high = mid - 1;
        }
        return low;
    }
}
